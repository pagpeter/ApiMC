ApiMC - API for namemc.com

Endpoints

/api/name/{name}

Example: GET /api/name/notch

Response data:

{
	"name": "notch",
	"uuid": "069a79f4-44e9-4726-a5be-fca90e38aaf5",
	"views": 5789,
	"location": null,
	"socials": {
		"discord": "notch#0000",
		"twitter": "notch"
	}
}

Code snippets

Python3 (requests)

 
from requests import get

name = "notch"
url = f"https://api.com/api/name/{name}"

r = get(url)

# handle error if non-200 response
if not r.ok: 
    exit("Error! Couldn't parse data")

data = r.json()
print(data)

Golang

 
package main

import "net/http"
import "fmt"

func main() {
    name := "notch"
    url := "https://api.com/api/name/" + name

    // Don't ignore errors in production
    resp, _ := http.Get(url)

    // Handle response
    defer resp.Body.Close()
    body, _ := io.ReadAll(resp.Body)
	
    fmt.Println(string(body))
}
